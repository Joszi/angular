Structural Directives

ng-if


<div *ngIf="book">
  <p>{{book.description}}</p>
</div>

Removes element from the DOM if book not present (this prevents errors - undefined.book is not valid).

ng-for


<ul>
  <li *ngFor="let item of list">{{item.property}}</li>
</ul>

Creates an element for every item of the collection.

Binding to custom events


@Component({
  selector: 'parent',
  template: '<child (message)="onChildClick($event)">'
})
export class ParentComponent {
  onChildClick(text: string): void {
    console.log('child gave me:', text);
  }
}

@Component({
  selector: 'child',
  template: '<button (click)="onButtonClick()">Click me</button>',
} as Component)
export class ChildComponent {
  @Output() message = new EventEmitter<string>();

  onButtonClick(): void {
    this.message.emit('value of the event');
  }
}
