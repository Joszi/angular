Http

The Angular HttpClient communicates with the server using a familiar HTTP request/response protocol.

The HttpClient is provided by the @angular/common/http library.

https://angular.io/guide/http

Http configuration

  • Importing HttpClientModule
  • Injecting HttpClient service

Importing HttpClientModule


// import
import { HttpClientModule } from '@angular/common/http';

// module configuration
@NgModule({
    imports: [HttpClientModule],
    declarations: [...],
    bootstrap: [...],
    providers: [...]
})
export class AppModule {}

Injecting HttpClient service


// import
import {HttpClient} from '@angular/common/http';

// injection
constructor(private http: HttpClient) {}

Http GET


this.http.get(this.requestUrl + id)
      .catch((error: any) => {
        console.log(error);
        return Observable.throw(error);
      });

Http PUT


this.http.put(this.requestUrl + id, requestBody)

Http POST


this.http.post(this.requestUrl, requestBody);
